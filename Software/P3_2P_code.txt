// =====================================================
// SISTEMA COMPLETO: Cámara (YUV422 SVGA) + Webhook(n8n)
//                   + Sensor IR + Motor DC (DRV8873) + Servo
// ESP32-S3 / Arduino-ESP32 core 2.0.x
//
// Flujo:
//   - Motor DC gira (vel 70) sin parar.
//   - IR detecta objeto (HIGH) -> para motor, ESPERA (delay enfoque),
//     toma foto y envía a n8n.
//   - Respuesta del webhook:
//       "verde"    -> servo NO se mueve, motor reanuda.
//       "amarillo" -> servo 90° (5 s y vuelve a 0), motor reanuda.
//       "rojo"     -> servo 180° (5 s y vuelve a 0), motor reanuda.
//   - Motor sigue hasta un NUEVO evento de detección.
//
// LEDC:  timer0/canal0 = cámara | timer1/canales2,3 = motor | timer2/canal4 = servo
// =====================================================

// ============== WEBHOOK (n8n en tu notebook) ==============http://192.168.100.2:5678/webhook/572b7ae6-a35c-47cf-968a-02e008a7816e
const char* WEBHOOK_URL =
    "http://10.47.199.114:5678/webhook/"
    "572b7ae6-a35c-47cf-968a-02e008a7816e";
#define WH_BOUNDARY "----esp32s3boundary7d3a91fED"

// ============== CALIDAD FOTO WEBHOOK ==============
// Calidad JPEG de la foto que se manda a n8n (0-100, mayor = mejor).
const int CALIDAD_WEBHOOK = 80;

// ============== SENSOR IR ==============
#define IR_PIN  48        // HIGH = objeto detectado

// ============== MOTOR DC (DRV8873) ==============
#define IN1     5
#define IN2     4
#define NFAULT  17
const int CANAL_IN1 = 2;     // canal LEDC 2 -> timer 1
const int CANAL_IN2 = 3;     // canal LEDC 3 -> timer 1
const int FREQ_PWM  = 20000;
const int RES_PWM   = 8;
const int VELOCIDAD = 100;

// ============== SERVO ==============
#define SERVO_PIN 43
Servo gateServo;

// ============== CÁMARA (pines) ==============
#define XCLK_GPIO_NUM   13
#define SIOD_GPIO_NUM    9
#define SIOC_GPIO_NUM   10
#define Y9_GPIO_NUM     47
#define Y8_GPIO_NUM     21
#define Y7_GPIO_NUM     38
#define Y6_GPIO_NUM     39

#define VSYNC_GPIO_NUM  12
#define HREF_GPIO_NUM   11
#define PCLK_GPIO_NUM   14

// ============== ESTADO ==============
bool objetoAntes = false;
bool servoActivo = false;
unsigned long tServo = 0;
const unsigned long SERVO_HOLD = 5000;    // 5 s

bool camaraOK = false, servoOK = false, wifiOK = false;

// =====================================================
// BASE64
// =====================================================
static void b64_encode(const uint8_t *src, size_t len, char *dst) {
  static const char tbl[] =
    "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    "abcdefghijklmnopqrstuvwxyz0123456789+/";
  size_t i = 0, j = 0;
  while (i < len) {
    size_t rem = len - i;
    uint8_t a = src[i++];
    uint8_t b = (rem > 1) ? src[i++] : 0;
    uint8_t c = (rem > 2) ? src[i++] : 0;
    dst[j++] = tbl[a >> 2];
    dst[j++] = tbl[((a & 0x03) << 4) | (b >> 4)];
    dst[j++] = (rem > 1) ? tbl[((b & 0x0f) << 2) | (c >> 6)] : '=';
    dst[j++] = (rem > 2) ? tbl[c & 0x3f] : '=';
  }
  dst[j] = '\0';
}

// =====================================================
// CONECTAR WIFI (busca por subcadena y usa el SSID exacto)
// =====================================================
bool conectarWiFi() {
  WiFi.mode(WIFI_STA);
  WiFi.disconnect(true);
  delay(200);

  Serial.println("[WIFI] escaneando...");
  int n = WiFi.scanNetworks();
  int idx = -1;
  for (int i = 0; i < n; i++)
    if (WiFi.SSID(i).indexOf(SSID_BUSCAR) >= 0) { idx = i; break; }

  if (idx < 0) { Serial.printf("[WIFI] no encontre red con '%s'\n", SSID_BUSCAR); return false; }

  String ssidReal = WiFi.SSID(idx);
  Serial.printf("[WIFI] conectando a [%s]", ssidReal.c_str());
  WiFi.begin(ssidReal.c_str(), WIFI_PASS);

  unsigned long t0 = millis();
  while (WiFi.status() != WL_CONNECTED && millis() - t0 < 20000) { delay(400); Serial.print("."); }
  Serial.println();
  return (WiFi.status() == WL_CONNECTED);
}

// =====================================================
// ENVÍO AL WEBHOOK (captura YUV422 -> JPEG por software)
// =====================================================
static bool sendToWebhook(String &response_out) {
  camera_fb_t *fb = esp_camera_fb_get();
  if (!fb) { Serial.println("[WH] fallo captura"); return false; }

  uint8_t *jpg_buf = NULL; size_t jpg_len = 0;
  bool ok = frame2jpg(fb, CALIDAD_WEBHOOK, &jpg_buf, &jpg_len);   // calidad arriba
  esp_camera_fb_return(fb);
  if (!ok || !jpg_buf) { Serial.println("[WH] fallo JPEG"); return false; }
  Serial.printf("[WH] JPEG %u bytes\n", jpg_len);

  size_t b64_chars = ((jpg_len + 2) / 3) * 4;
  char *b64 = (char*)ps_malloc(b64_chars + 1);
  if (!b64) b64 = (char*)malloc(b64_chars + 1);
  if (!b64) { free(jpg_buf); Serial.println("[WH] sin memoria b64"); return false; }
  b64_encode(jpg_buf, jpg_len, b64);

  String part1 =
    "--" WH_BOUNDARY "\r\n"
    "Content-Disposition: form-data; name=\"data\"\r\n\r\n";
  String part2 =
    "\r\n--" WH_BOUNDARY "\r\n"
    "Content-Disposition: form-data; name=\"data\"; filename=\"capture.jpg\"\r\n"
    "Content-Type: image/jpeg\r\n\r\n";
  String tail = "\r\n--" WH_BOUNDARY "--\r\n";

  size_t total = part1.length() + b64_chars + part2.length() + jpg_len + tail.length();
  uint8_t *body = (uint8_t*)ps_malloc(total);
  if (!body) body = (uint8_t*)malloc(total);
  if (!body) { free(b64); free(jpg_buf); Serial.println("[WH] sin memoria body"); return false; }

  size_t off = 0;
  memcpy(body+off, part1.c_str(), part1.length()); off += part1.length();
  memcpy(body+off, b64, b64_chars);                off += b64_chars;
  memcpy(body+off, part2.c_str(), part2.length()); off += part2.length();
  memcpy(body+off, jpg_buf, jpg_len);              off += jpg_len;
  memcpy(body+off, tail.c_str(), tail.length());   off += tail.length();
  free(b64); free(jpg_buf);

  WiFiClient client; HTTPClient http;
  bool success = false;
  if (http.begin(client, WEBHOOK_URL)) {
    http.addHeader("Content-Type", "multipart/form-data; boundary=" WH_BOUNDARY);
    http.setTimeout(60000);
    int code = http.POST(body, total);
    if (code > 0) {
      Serial.printf("[WH] HTTP %d\n", code);
      response_out = http.getString();
      Serial.println("[WH] resp: " + response_out);
      success = true;
    } else {
      Serial.printf("[WH] error: %s\n", http.errorToString(code).c_str());
    }
    http.end();
  } else {
    Serial.println("[WH] http.begin() fallo");
  }
  free(body);
  return success;
}

// =====================================================
// PARSEAR COLOR del JSON  [{"color":"rojo",...}]
// =====================================================
static String parseColor(const String &json) {
  int k  = json.indexOf("\"color\"");      if (k  < 0) return "";
  int c  = json.indexOf(':', k);           if (c  < 0) return "";
  int q1 = json.indexOf('"', c);           if (q1 < 0) return "";
  int q2 = json.indexOf('"', q1 + 1);      if (q2 < 0) return "";
  String val = json.substring(q1 + 1, q2);
  val.toLowerCase(); val.trim();
  return val;
}

// =====================================================
// MOTOR
// =====================================================
void motorGira() { ledcWrite(CANAL_IN1, VELOCIDAD); ledcWrite(CANAL_IN2, 0); }
void motorPara() { ledcWrite(CANAL_IN1, 0);         ledcWrite(CANAL_IN2, 0); }

// =====================================================
// ACCIÓN SEGÚN COLOR
// =====================================================
void procesarColor(const String &color) {
  if (color == "verde") {
    Serial.println("[ACCION] verde -> servo no se mueve");
    gateServo.write(0);
    servoActivo = false;
  } else if (color == "amarillo") {
    Serial.println("[ACCION] amarillo -> servo 90");
    gateServo.write(90);
    servoActivo = true; tServo = millis();
  } else if (color == "rojo") {
    Serial.println("[ACCION] rojo -> servo 180");
    gateServo.write(180);
    servoActivo = true; tServo = millis();
  } else {
    Serial.printf("[ACCION] color '%s' desconocido -> sin accion\n", color.c_str());
  }
}

// =====================================================
// SETUP
// =====================================================
void setup() {
  Serial.begin(115200);
  delay(1000);
  Serial.println("\n=== ARRANQUE ===");

  pinMode(IR_PIN, INPUT_PULLUP);

  // MOTOR DC primero
  pinMode(NFAULT, INPUT_PULLUP);
  ledcSetup(CANAL_IN1, FREQ_PWM, RES_PWM);
  ledcSetup(CANAL_IN2, FREQ_PWM, RES_PWM);
  ledcAttachPin(IN1, CANAL_IN1);
  ledcAttachPin(IN2, CANAL_IN2);
  motorGira();
  Serial.println("[MOTOR] OK -> girando");

  // SERVO
  ESP32PWM::allocateTimer(2);
  ESP32PWM::allocateTimer(3);
  gateServo.setPeriodHertz(50);
  int canalServo = gateServo.attach(SERVO_PIN, 500, 2500);
  if (canalServo != 0) { servoOK = true; gateServo.write(0); Serial.printf("[SERVO] OK (canal %d)\n", canalServo); }
  else                 { servoOK = false; Serial.println("[SERVO] FALLO"); }

  // CÁMARA (YUV422 SVGA - buena calidad y colores correctos)
  pinMode(SIOD_GPIO_NUM, INPUT_PULLUP);
  pinMode(SIOC_GPIO_NUM, INPUT_PULLUP);
  camera_config_t cfg = {};
  cfg.ledc_channel = LEDC_CHANNEL_0;
  cfg.ledc_timer   = LEDC_TIMER_0;
  cfg.pin_d0 = Y2_GPIO_NUM; cfg.pin_d1 = Y3_GPIO_NUM;
  cfg.pin_d2 = Y4_GPIO_NUM; cfg.pin_d3 = Y5_GPIO_NUM;
  cfg.pin_d4 = Y6_GPIO_NUM; cfg.pin_d5 = Y7_GPIO_NUM;
  cfg.pin_d6 = Y8_GPIO_NUM; cfg.pin_d7 = Y9_GPIO_NUM;
  cfg.pin_xclk = XCLK_GPIO_NUM;  cfg.pin_pclk = PCLK_GPIO_NUM;
  cfg.pin_vsync = VSYNC_GPIO_NUM; cfg.pin_href = HREF_GPIO_NUM;
  cfg.pin_sccb_sda = SIOD_GPIO_NUM; cfg.pin_sccb_scl = SIOC_GPIO_NUM;
  cfg.pin_pwdn = -1; cfg.pin_reset = -1;

  cfg.xclk_freq_hz = 20000000;
  cfg.pixel_format = PIXFORMAT_YUV422;     // color nativo correcto
  cfg.frame_size   = FRAMESIZE_SVGA;       // 800x600 (baja a VGA si da fb NULL)
  cfg.fb_count     = 1;
  cfg.grab_mode    = CAMERA_GRAB_WHEN_EMPTY;
  cfg.fb_location  = CAMERA_FB_IN_PSRAM;

  esp_err_t errCam = esp_camera_init(&cfg);
  if (errCam == ESP_OK) { camaraOK = true; Serial.println("[CAMARA] OK (YUV422 SVGA)"); }
  else                  { camaraOK = false; Serial.printf("[CAMARA] FALLO 0x%x\n", errCam); }

  // WIFI
  wifiOK = conectarWiFi();
  if (wifiOK) { Serial.print("[WIFI] OK -> IP: "); Serial.println(WiFi.localIP()); }
  else        { Serial.println("[WIFI] FALLO (el webhook no funcionara)"); }

  // RESUMEN
  Serial.println("\n---------- RESUMEN ----------");
  Serial.printf("  Camara : %s\n", camaraOK ? "OK" : "FALLO");
  Serial.printf("  Servo  : %s\n", servoOK  ? "OK" : "FALLO");
  Serial.printf("  WiFi   : %s\n", wifiOK   ? "OK" : "FALLO");
  Serial.println("-----------------------------");
  Serial.println("=== Sistema listo. Motor girando. ===\n");
}

// =====================================================
// LOOP
// =====================================================
void loop() {
  bool objeto = (digitalRead(IR_PIN) == HIGH);   // HIGH = detectado

  if (objeto && !objetoAntes) {
    Serial.println("\n[IR] objeto detectado -> parar y fotografiar");
    motorPara();

    // ============================================================
    // >>> DELAY DE ENFOQUE (AJUSTABLE) <
    // Espera a que la caja quede quieta/enfocada antes de la foto.
    // Sube o baja este 2000 (en milisegundos) durante tus pruebas.
    delay(2000);   // <<<<<< AQUI: 2 segundos antes de la foto
    // ============================================================

    String respuesta;
    bool ok = sendToWebhook(respuesta);
    if (ok) {
      String color = parseColor(respuesta);
      Serial.println("[IR] color: " + color);
      procesarColor(color);
    } else {
      Serial.println("[IR] fallo el envio");
    }
    motorGira();
  }
  objetoAntes = objeto;

  if (servoActivo && (millis() - tServo >= SERVO_HOLD)) {
    gateServo.write(0);
    servoActivo = false;
    Serial.println("[SERVO] vuelve a 0");
  }

  delay(20);
}
