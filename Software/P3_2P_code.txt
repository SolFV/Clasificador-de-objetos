// ================= STREAM =================
#define PART_BOUNDARY "123456789000000000000987654321"

static const char* STREAM_CONTENT_TYPE =
"multipart/x-mixed-replace;boundary=" PART_BOUNDARY;

static const char* STREAM_BOUNDARY =
"\r\n--" PART_BOUNDARY "\r\n";

static const char* STREAM_PART =
"Content-Type: image/jpeg\r\nContent-Length: %u\r\n\r\n";

// =====================================================
// BASE64 ENCODE  (sin librerías externas)
// =====================================================
static void b64_encode(const uint8_t *src, size_t len, char *dst)
{
    static const char tbl[] =
        "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
        "abcdefghijklmnopqrstuvwxyz"
        "0123456789+/";

    size_t i = 0, j = 0;
    while (i < len) {
        size_t  rem = len - i;
        uint8_t a = src[i++];
        uint8_t b = (rem > 1) ? src[i++] : 0;
        uint8_t c = (rem > 2) ? src[i++] : 0;

        dst[j++] = tbl[a >> 2];
        dst[j++] = tbl[((a & 0x03) << 4) | (b >> 4)];
        dst[j++] = (rem > 1) ? tbl[((b & 0x0f) << 2) | (c >> 6)] : '=';
        dst[j++] = (rem > 2) ? tbl[c & 0x3f] : '=';
    }
    dst[j] = '\0';
}

// =====================================================
// SEND TO WEBHOOK
//   1. captura frame
//   2. convierte a JPEG (calidad 40)
//   3. codifica en base64
//   4. POST multipart/form-data con DOS partes:
//        - campo texto  "data" = base64   -> $json.body.data  (AI Agent)
//        - archivo      "data" = JPEG bin -> $binary.data     (Subir a Drive)
//   5. deja la respuesta del workflow en response_out
// =====================================================

// Boundary para el multipart (cualquier string que no aparezca en los datos)
#define WH_BOUNDARY "----esp32s3boundary7d3a91fED"

static bool sendToWebhook(String &response_out)
{
    // 1. Frame
    camera_fb_t *fb = esp_camera_fb_get();
    if (!fb) {
        Serial.println("[WH] fallo captura");
        return false;
    }

    // 2. JPEG
    uint8_t *jpg_buf = NULL;
    size_t   jpg_len = 0;
    bool ok = frame2jpg(fb, 40, &jpg_buf, &jpg_len);
    esp_camera_fb_return(fb);

    if (!ok || !jpg_buf) {
        Serial.println("[WH] fallo JPEG");
        return false;
    }
    Serial.printf("[WH] JPEG: %u bytes\n", jpg_len);

    // 3. base64 del JPEG (para el campo de texto que lee el AI Agent)
    size_t b64_chars = ((jpg_len + 2) / 3) * 4;
    char *b64 = (char*)ps_malloc(b64_chars + 1);
    if (!b64) b64 = (char*)malloc(b64_chars + 1);
    if (!b64) {
        free(jpg_buf);
        Serial.println("[WH] sin memoria (b64)");
        return false;
    }
    b64_encode(jpg_buf, jpg_len, b64);

    // 4. Construir las partes del multipart
    //    PARTE 1: campo texto "data" = base64
    String part1 =
        "--" WH_BOUNDARY "\r\n"
        "Content-Disposition: form-data; name=\"data\"\r\n\r\n";

    //    PARTE 2: archivo "data" = imagen JPEG binaria
    String part2 =
        "\r\n--" WH_BOUNDARY "\r\n"
        "Content-Disposition: form-data; name=\"data\"; filename=\"capture.jpg\"\r\n"
        "Content-Type: image/jpeg\r\n\r\n";

    //    Cierre
    String tail = "\r\n--" WH_BOUNDARY "--\r\n";

    size_t total =
        part1.length() + b64_chars +
        part2.length() + jpg_len +
        tail.length();

    // 5. Alocar el cuerpo completo en PSRAM
    uint8_t *body = (uint8_t*)ps_malloc(total);
    if (!body) body = (uint8_t*)malloc(total);
    if (!body) {
        free(b64);
        free(jpg_buf);
        Serial.println("[WH] sin memoria (body)");
        return false;
    }

    // 6. Ensamblar el cuerpo: part1 + base64 + part2 + jpeg + tail
    size_t off = 0;
    memcpy(body + off, part1.c_str(), part1.length()); off += part1.length();
    memcpy(body + off, b64,           b64_chars);       off += b64_chars;
    memcpy(body + off, part2.c_str(), part2.length()); off += part2.length();
    memcpy(body + off, jpg_buf,       jpg_len);          off += jpg_len;
    memcpy(body + off, tail.c_str(),  tail.length());    off += tail.length();

    free(b64);
    free(jpg_buf);

    // 7. POST HTTPS
    WiFiClientSecure tls;
    tls.setInsecure();          // sin verificar cert (OK para pruebas)

    HTTPClient http;
    bool success = false;

    if (http.begin(tls, WEBHOOK_URL)) {
        http.addHeader("Content-Type",
            "multipart/form-data; boundary=" WH_BOUNDARY);
        http.setTimeout(60000); // 60s, el modelo AI puede tardar

        int code = http.POST(body, total);
        if (code > 0) {
            Serial.printf("[WH] HTTP %d\n", code);
            response_out = http.getString();
            Serial.println("[WH] resp: " + response_out);
            success = true;
        } else {
            Serial.printf("[WH] error: %s\n",
                http.errorToString(code).c_str());
        }
        http.end();
    } else {
        Serial.println("[WH] http.begin() fallo");
    }

    free(body);
    return success;
}

// =====================================================
// PARSEAR COLOR de la respuesta de n8n
// Respuesta tipo:  [{"color":"rojo","descripcion":"..."}]
// =====================================================
static String parseColor(const String &json)
{
    int k = json.indexOf("\"color\"");
    if (k < 0) return "";
    int c = json.indexOf(':', k);
    if (c < 0) return "";
    int q1 = json.indexOf('"', c);
    if (q1 < 0) return "";
    int q2 = json.indexOf('"', q1 + 1);
    if (q2 < 0) return "";

    String val = json.substring(q1 + 1, q2);
    val.toLowerCase();
    val.trim();
    return val;
}

// =====================================================
// MOVER SERVO segun el color
//   rojo     -> 90 grados,  espera 10s, vuelve a 0
//   amarillo -> 180 grados, espera 10s, vuelve a 0
//   verde    -> no se mueve
// =====================================================
static void moveServoForColor(const String &color)
{
    int angle;
    if (color == "rojo")          angle = 90;
    else if (color == "amarillo") angle = 180;
    else {
        Serial.printf("[SERVO] color '%s' -> no mueve\n", color.c_str());
        return;
    }

    Serial.printf("[SERVO] %s -> %d grados\n", color.c_str(), angle);
    gateServo.write(angle);
    delay(SERVO_HOLD);          // mantener 10 segundos
    gateServo.write(0);
    Serial.println("[SERVO] vuelve a 0 grados");
}

// =====================================================
// HANDLER  GET /send
// Captura, manda al webhook, responde al browser y
// luego mueve el servo segun el color recibido.
// =====================================================
static esp_err_t send_handler(httpd_req_t *req)
{
    String response;
    bool ok = sendToWebhook(response);

    httpd_resp_set_type(req, "application/json");

    esp_err_t res;
    if (ok) {
        res = httpd_resp_send(req,
            response.c_str(), response.length());
    } else {
        const char *err = "{\"error\":\"fallo el envio\"}";
        httpd_resp_set_status(req, "500 Internal Server Error");
        res = httpd_resp_send(req, err, strlen(err));
    }

    // Ya respondimos al navegador; ahora movemos el servo.
    // (El delay de 10s NO afecta al browser porque ya recibio su respuesta.)
    if (ok) {
        String color = parseColor(response);
        moveServoForColor(color);
    }

    return res;
}

// =====================================================
// CAPTURE
// =====================================================
static esp_err_t capture_handler(httpd_req_t *req)
{
    camera_fb_t *fb = esp_camera_fb_get();

    if (!fb) {
        Serial.println("fb NULL");
        httpd_resp_send_500(req);
        return ESP_FAIL;
    }

    uint8_t *jpg_buf = NULL;
    size_t jpg_len = 0;

    bool ok = frame2jpg(
        fb,
        80,
        &jpg_buf,
        &jpg_len
    );

    esp_camera_fb_return(fb);

    if (!ok) {
        Serial.println("frame2jpg fallo");
        httpd_resp_send_500(req);
        return ESP_FAIL;
    }

    httpd_resp_set_type(req, "image/jpeg");

    esp_err_t res = httpd_resp_send(
        req,
        (const char*)jpg_buf,
        jpg_len
    );

    free(jpg_buf);

    return res;
}

// =====================================================
// STREAM
// =====================================================
static esp_err_t stream_handler(httpd_req_t *req)
{
    camera_fb_t *fb = NULL;
    esp_err_t res = ESP_OK;
    char part_buf[64];

    httpd_resp_set_type(req, STREAM_CONTENT_TYPE);

    while (true) {

        fb = esp_camera_fb_get();

        if (!fb) {
            Serial.println("fb NULL");
            delay(20);
            continue;
        }

        uint8_t *jpg_buf = NULL;
        size_t jpg_len = 0;

        bool ok = frame2jpg(
            fb,
            70,
            &jpg_buf,
            &jpg_len
        );

        esp_camera_fb_return(fb);

        if (!ok) {
            Serial.println("frame2jpg fail");
            continue;
        }

        size_t hlen = snprintf(
            part_buf,
            sizeof(part_buf),
            STREAM_PART,
            jpg_len
        );

        res = httpd_resp_send_chunk(
            req,
            STREAM_BOUNDARY,
            strlen(STREAM_BOUNDARY)
        );

        if (res == ESP_OK)
            res = httpd_resp_send_chunk(
                req,
                part_buf,
                hlen
            );

        if (res == ESP_OK)
            res = httpd_resp_send_chunk(
                req,
                (const char*)jpg_buf,
                jpg_len
            );

        free(jpg_buf);

        if (res != ESP_OK)
            break;
    }

    return res;
}

// =====================================================
void startCameraServer()
{
    // ---- Servidor 1 (puerto 80): index, capture, send ----
    httpd_config_t config =
        HTTPD_DEFAULT_CONFIG();

    config.server_port = 80;
    config.ctrl_port   = 32768;
    config.stack_size  = 16384;   // SSL/TLS para /send necesita stack grande
    config.max_uri_handlers = 8;

    httpd_uri_t index_uri = {
        .uri="/",
        .method=HTTP_GET,
        .handler=index_handler
    };

    httpd_uri_t capture_uri = {
        .uri="/capture",
        .method=HTTP_GET,
        .handler=capture_handler
    };

    httpd_uri_t send_uri = {
        .uri="/send",
        .method=HTTP_GET,
        .handler=send_handler
    };

    httpd_start(
        &server_httpd,
        &config
    );

    httpd_register_uri_handler(
        server_httpd,
        &index_uri
    );

    httpd_register_uri_handler(
        server_httpd,
        &capture_uri
    );

    httpd_register_uri_handler(
        server_httpd,
        &send_uri
    );

    // ---- Servidor 2 (puerto 81): SOLO el stream ----
    // En su propia task, asi el bucle infinito del stream
    // no bloquea las peticiones a /send y /capture.
    config.server_port = 81;
    config.ctrl_port   = 32769;
    config.stack_size  = 8192;

    httpd_uri_t stream_uri = {
        .uri="/stream",
        .method=HTTP_GET,
        .handler=stream_handler
    };

    httpd_start(
        &stream_httpd,
        &config
    );

    httpd_register_uri_handler(
        stream_httpd,
        &stream_uri
    );
}

// =====================================================
void setup()
{
    Serial.begin(115200);
    delay(2000);

    pinMode(SIOD_GPIO_NUM, INPUT_PULLUP);
    pinMode(SIOC_GPIO_NUM, INPUT_PULLUP);

    camera_config_t config = {};

    config.ledc_channel =
        LEDC_CHANNEL_0;

    config.ledc_timer =
        LEDC_TIMER_0;

    config.pin_d0 = Y2_GPIO_NUM;
    config.pin_d1 = Y3_GPIO_NUM;
    config.pin_d2 = Y4_GPIO_NUM;
    config.pin_d3 = Y5_GPIO_NUM;
    config.pin_d4 = Y6_GPIO_NUM;
    config.pin_d5 = Y7_GPIO_NUM;
    config.pin_d6 = Y8_GPIO_NUM;
    config.pin_d7 = Y9_GPIO_NUM;

    config.pin_xclk = XCLK_GPIO_NUM;
    config.pin_pclk = PCLK_GPIO_NUM;
    config.pin_vsync = VSYNC_GPIO_NUM;
    config.pin_href = HREF_GPIO_NUM;

    config.pin_sccb_sda = SIOD_GPIO_NUM;
    config.pin_sccb_scl = SIOC_GPIO_NUM;

    config.pin_pwdn = -1;
    config.pin_reset = -1;

    config.xclk_freq_hz = 24000000;

    config.pixel_format = PIXFORMAT_RGB565;
    config.frame_size = FRAMESIZE_QVGA;

    config.fb_count = 1;
    config.fb_location =
        CAMERA_FB_IN_PSRAM;

    esp_err_t err =
        esp_camera_init(&config);

    if (err != ESP_OK) {
        Serial.printf(
            "init fail 0x%x\n",
            err
        );
        return;
    }

    Serial.println("Camara OK RGB565");

    // ---- Servo ----
    // Usamos timers LEDC 1,2,3 para el servo.
    // El timer 0 lo usa la camara (XCLK), por eso NO lo tocamos.
    ESP32PWM::allocateTimer(1);
    ESP32PWM::allocateTimer(2);
    ESP32PWM::allocateTimer(3);
    gateServo.setPeriodHertz(50);              // servos estandar = 50 Hz
    gateServo.attach(SERVO_PIN, 500, 2500);    // ancho de pulso 0.5ms-2.5ms
    gateServo.write(0);                        // posicion inicial
    Serial.println("Servo OK en GPIO 43");

    WiFi.begin(
        WIFI_SSID,
        WIFI_PASS
    );

    while (
        WiFi.status()
        != WL_CONNECTED
    ) {
        delay(300);
        Serial.print(".");
    }

    Serial.println();
    Serial.println(
        WiFi.localIP()
    );

    startCameraServer();
}

void loop()
{
    delay(1000);
}
