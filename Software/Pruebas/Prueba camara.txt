httpd_handle_t server_httpd = NULL;

#define PART_BOUNDARY "123456789000000000000987654321"
static const char* STREAM_CONTENT_TYPE = "multipart/x-mixed-replace;boundary=" PART_BOUNDARY;
static const char* STREAM_BOUNDARY     = "\r\n--" PART_BOUNDARY "\r\n";
static const char* STREAM_PART         = "Content-Type: image/jpeg\r\nContent-Length: %u\r\n\r\n";

// ---------- Capturar UNA foto ----------
static esp_err_t capture_handler(httpd_req_t *req) {
  Serial.println("[capture] cliente conectado");
  camera_fb_t *fb = esp_camera_fb_get();
  if (!fb) {
    Serial.println("[capture] fb_get devolvio NULL");
    httpd_resp_send_500(req);
    return ESP_FAIL;
  }
  Serial.printf("[capture] frame %u bytes\n", (unsigned)fb->len);
  httpd_resp_set_type(req, "image/jpeg");
  httpd_resp_set_hdr(req, "Content-Disposition", "inline; filename=capture.jpg");
  httpd_resp_set_hdr(req, "Access-Control-Allow-Origin", "*");
  httpd_resp_set_hdr(req, "Cache-Control", "no-cache, no-store, must-revalidate");
  esp_err_t res = httpd_resp_send(req, (const char*)fb->buf, fb->len);
  esp_camera_fb_return(fb);
  return res;
}

// ---------- Stream MJPEG ----------
static esp_err_t stream_handler(httpd_req_t *req) {
  Serial.println("[stream] cliente conectado");
  camera_fb_t *fb = NULL;
  esp_err_t res = ESP_OK;
  char hdr[64];

  res = httpd_resp_set_type(req, STREAM_CONTENT_TYPE);
  if (res != ESP_OK) return res;
  httpd_resp_set_hdr(req, "Access-Control-Allow-Origin", "*");
  httpd_resp_set_hdr(req, "Cache-Control", "no-cache, no-store, must-revalidate");
  httpd_resp_set_hdr(req, "Pragma", "no-cache");

  int frames = 0;
  int nulls  = 0;

  while (true) {
    fb = esp_camera_fb_get();
    if (!fb) {
      nulls++;
      if (nulls > 10) { Serial.println("[stream] demasiados null, salgo"); break; }
      vTaskDelay(pdMS_TO_TICKS(100));
      continue;
    }
    nulls = 0;

    if (fb->format != PIXFORMAT_JPEG) {
      esp_camera_fb_return(fb);
      vTaskDelay(pdMS_TO_TICKS(50));
      continue;
    }

    size_t hlen = snprintf(hdr, 64, STREAM_PART, fb->len);
    res = httpd_resp_send_chunk(req, STREAM_BOUNDARY, strlen(STREAM_BOUNDARY));
    if (res == ESP_OK) res = httpd_resp_send_chunk(req, hdr, hlen);
    if (res == ESP_OK) res = httpd_resp_send_chunk(req, (const char*)fb->buf, fb->len);
    esp_camera_fb_return(fb);

    if (res != ESP_OK) { Serial.printf("[stream] cliente cerro tras %d frames\n", frames); break; }
    frames++;
    if (frames % 10 == 0) Serial.printf("[stream] %d frames\n", frames);
    vTaskDelay(pdMS_TO_TICKS(100));
  }
  return res;
}

void startCameraServer() {
  httpd_config_t config = HTTPD_DEFAULT_CONFIG();
  config.server_port      = 80;
  config.stack_size       = 16384;
  config.max_uri_handlers = 8;

  httpd_uri_t u_index   = { .uri = "/",        .method = HTTP_GET, .handler = index_handler,   .user_ctx = NULL };
  httpd_uri_t u_capture = { .uri = "/capture", .method = HTTP_GET, .handler = capture_handler, .user_ctx = NULL };
  httpd_uri_t u_stream  = { .uri = "/stream",  .method = HTTP_GET, .handler = stream_handler,  .user_ctx = NULL };

  if (httpd_start(&server_httpd, &config) == ESP_OK) {
    httpd_register_uri_handler(server_httpd, &u_index);
    httpd_register_uri_handler(server_httpd, &u_capture);
    httpd_register_uri_handler(server_httpd, &u_stream);
    Serial.println("Servidor HTTP arrancado.");
  } else {
    Serial.println("Fallo httpd_start");
  }
}

bool initCamera() {
  esp_camera_deinit();
  delay(300);

  camera_config_t config = {};
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer   = LEDC_TIMER_0;
  config.pin_d0 = Y2_GPIO_NUM; config.pin_d1 = Y3_GPIO_NUM;
  config.pin_d2 = Y4_GPIO_NUM; config.pin_d3 = Y5_GPIO_NUM;
  config.pin_d4 = Y6_GPIO_NUM; config.pin_d5 = Y7_GPIO_NUM;
  config.pin_d6 = Y8_GPIO_NUM; config.pin_d7 = Y9_GPIO_NUM;
  config.pin_xclk     = XCLK_GPIO_NUM;    config.pin_pclk     = PCLK_GPIO_NUM;
  config.pin_vsync    = VSYNC_GPIO_NUM;   config.pin_href     = HREF_GPIO_NUM;
  config.pin_sccb_sda = SIOD_GPIO_NUM;    config.pin_sccb_scl = SIOC_GPIO_NUM;
  config.pin_pwdn     = PWDN_GPIO_NUM;    config.pin_reset    = RESET_GPIO_NUM;

  // Paso 1: init en RGB565 — fuerza reset completo del sensor
  // El OV5640 a veces arranca mal en JPEG directo
  config.xclk_freq_hz = 20000000;
  config.pixel_format = PIXFORMAT_RGB565;
  config.frame_size   = FRAMESIZE_QVGA;
  config.jpeg_quality = 12;
  config.fb_count     = 1;
  config.grab_mode    = CAMERA_GRAB_WHEN_EMPTY;
  config.fb_location  = CAMERA_FB_IN_PSRAM;

  esp_err_t err = esp_camera_init(&config);
  if (err != ESP_OK) {
    Serial.printf("ERROR init RGB565 (0x%x)\n", err);
    return false;
  }
  Serial.println("Init RGB565 OK — reseteando sensor...");

  // Captura algunos frames en RGB para despertar el sensor
  for (int i = 0; i < 5; i++) {
    camera_fb_t *fb = esp_camera_fb_get();
    if (fb) {
      Serial.printf("  RGB frame %d: %u bytes\n", i, (unsigned)fb->len);
      esp_camera_fb_return(fb);
    } else {
      Serial.printf("  RGB frame %d: NULL\n", i);
    }
    delay(150);
  }

  // Paso 2: switch a JPEG via sensor register
  sensor_t *s = esp_camera_sensor_get();
  if (s) {
    s->set_pixformat(s, PIXFORMAT_JPEG);
    s->set_framesize(s, FRAMESIZE_QVGA);
    s->set_quality(s, 12);
    s->set_brightness(s, 0);
    s->set_contrast(s, 0);
    s->set_saturation(s, 0);
    s->set_whitebal(s, 1);
    s->set_awb_gain(s, 1);
    s->set_wb_mode(s, 0);
    s->set_exposure_ctrl(s, 1);
    s->set_aec2(s, 0);
    s->set_gain_ctrl(s, 1);
    s->set_agc_gain(s, 0);
    s->set_gainceiling(s, (gainceiling_t)0);
    s->set_bpc(s, 0);
    s->set_wpc(s, 1);
    s->set_raw_gma(s, 1);
    s->set_lenc(s, 1);
    s->set_hmirror(s, 0);
    s->set_vflip(s, 0);
    Serial.println("Sensor configurado en JPEG.");
  }

  delay(500); // espera a que el sensor aplique el cambio de formato

  // Paso 3: reinit con JPEG y fb_count=2
  esp_camera_deinit();
  delay(200);

  config.pixel_format = PIXFORMAT_JPEG;
  config.fb_count     = 2;

  err = esp_camera_init(&config);
  if (err != ESP_OK) {
    Serial.printf("ERROR init JPEG (0x%x)\n", err);
    return false;
  }
  Serial.println("Camara OK en JPEG.");

  // Configura sensor de nuevo tras reinit
  s = esp_camera_sensor_get();
  if (s) {
    s->set_framesize(s, FRAMESIZE_QVGA);
    s->set_quality(s, 12);
    s->set_whitebal(s, 1);
    s->set_awb_gain(s, 1);
    s->set_exposure_ctrl(s, 1);
    s->set_gain_ctrl(s, 1);
    s->set_raw_gma(s, 1);
    s->set_lenc(s, 1);
  }

  // Warm-up final
  Serial.println("Warm-up JPEG...");
  int ok = 0;
  for (int i = 0; i < 10; i++) {
    camera_fb_t *fb = esp_camera_fb_get();
    if (fb) {
      Serial.printf("  jpeg frame %d: %u bytes\n", i, (unsigned)fb->len);
      esp_camera_fb_return(fb);
      ok++;
    } else {
      Serial.printf("  jpeg frame %d: NULL\n", i);
    }
    delay(150);
  }

  if (ok == 0) {
    Serial.println("ERROR: ningun frame JPEG valido en warm-up");
    return false;
  }

  Serial.printf("Warm-up OK (%d/%d frames validos)\n", ok, 10);
  return true;
}

void setup() {
  Serial.begin(115200);
  Serial.setDebugOutput(true);
  delay(2000);
  Serial.println("\n=== OV5640 streaming ===");

  if (!initCamera()) {
    Serial.println("Reintentando en 3s...");
    delay(3000);
    if (!initCamera()) {
      Serial.println("FATAL: camara no disponible");
      return;
    }
  }

  WiFi.mode(WIFI_STA);
  WiFi.begin(WIFI_SSID, WIFI_PASS);
  Serial.print("Conectando WiFi");
  int intentos = 0;
  while (WiFi.status() != WL_CONNECTED) {
    delay(300);
    Serial.print(".");
    if (++intentos > 40) {
      Serial.println("\nERROR: WiFi timeout");
      return;
    }
  }
  Serial.println();
  Serial.print(">>> http://");
  Serial.print(WiFi.localIP());
  Serial.println("/");

  startCameraServer();
}

void loop() {
  delay(10000);
}
